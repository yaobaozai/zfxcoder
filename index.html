angularjs高级程序设计读书笔记系列之一

发表于 2017-03-30

不跑步怎么减肥
暂时应该只能够自己复习用， 敬请谅解。
先从第六章开始，在这一章里，作者完成了三个功能。按类名分类，分页，选中高亮。

前两个功能跟过滤器紧密相连
我把完整的代码贴在这里

angular.module(‘customFilters’,[]).filter(‘unique’,function(){
return function(data,propertyName){
// 类型判断 突出一个严谨
// 空数组用来存放 结果与临时判断条件用的
// 遍历数据
// 命名val
// 如果不存在
// 则让keys的val为true
// 添加进结果数组
// 返回出去
// 类型不满足就直接返回原数据
if(angular.isArray(data) && angular.isString(propertyName)){
var results = [];
var keys = [];
for(var i = 0; i < data.length; i++){
var val = data[i][propertyName];
if(angular.isUndefined(keys[val])){
keys[val] = true;
results.push(val);
}
}
return results;
}else{
return data;
}
}
})
.filter(‘range’,function($filter){
return function(data,page,size){
//类型判断
if(angular.isArray(data) && angular.isNumber(page) && angular.isNumber(size)){
//开始的索引（假设一页三条数据，那么当索引为2的时候应该从3开始）
var start_index = (page - 1) * size;
//如果开始的索引比数据的长度大（即超过了数据的总量，返回空数组）
if(data.length < start_index){
return [];
}else{
//否则的话
//这个概念难以理解，我认为是作者的问题
//前面他考虑读者的基础而介绍基础 这里突然却草草的带过
//原书印刷还有错误，我这里已经更正过来了，导致我在这里卡了很久。
//其实这个一点都不绕
//截取一个分页
return $filter(‘limitTo’)(data.splice(start_index, size));
}
}else{
return data;
}
}
})
.filter(‘pageCount’,function(){
//这个比上面好理解多了，就是计算
return function(data,size){
if(angular.isArray(data)){
var result = [];
for(var i = 0; i < Math.ceil(data.length / size); i++){
result.push(i);
}
return result;
}else{
return data;
}
}
})
